#import <Foundation/Foundation>;

@interface ApiResult : NSObject

//数据
@property (nonatomic, strong) PageInfo data;
//状态码
@property (nonatomic, strong) NSInteger code;
//消息
@property (nonatomic, assign) NSString msg;


@end



#import <Foundation/Foundation>;

@interface PageInfo : NSObject

//当前页
@property (nonatomic, strong) NSInteger pageNum;
//每页的数量
@property (nonatomic, strong) NSInteger pageSize;
//当前页的数量
@property (nonatomic, strong) NSInteger size;
//当前页面第一个元素在数据库中的行号
@property (nonatomic, strong) long startRow;
//当前页面最后一个元素在数据库中的行号
@property (nonatomic, strong) long endRow;
//总页数
@property (nonatomic, strong) NSInteger pages;
//前一页
@property (nonatomic, strong) NSInteger prePage;
//下一页
@property (nonatomic, strong) NSInteger nextPage;
//是否为第一页
@property (nonatomic, strong) BOOL isFirstPage;
//是否为最后一页
@property (nonatomic, strong) BOOL isLastPage;
//是否有前一页
@property (nonatomic, strong) BOOL hasPreviousPage;
//是否有下一页
@property (nonatomic, strong) BOOL hasNextPage;
//导航页码数
@property (nonatomic, strong) NSInteger navigatePages;
//所有导航页号
@property (nonatomic, strong) int[] navigatepageNums;
//导航条上的第一页
@property (nonatomic, strong) NSInteger navigateFirstPage;
//导航条上的最后一页
@property (nonatomic, strong) NSInteger navigateLastPage;
//总记录数
@property (nonatomic, strong) long total;
//结果集
@property (nonatomic, strong) Document[] list;


@end



#import <Foundation/Foundation>;

@interface Document : NSObject

//主键
@property (nonatomic, strong) long id;
//文档标题
@property (nonatomic, assign) NSString title;
//文档内容
@property (nonatomic, assign) NSString content;


@end
